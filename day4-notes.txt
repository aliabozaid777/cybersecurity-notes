# Start Web Server
python3 -m http.server 8000

# Scan Open Ports
nmap 192.168.100.8

# Detect Service Version
nmap -sV 192.168.100.8

# Download Files
wget http://192.168.100.8:8000/secret.txt
curl -O http://192.168.100.8:8000/secret.txt

# Banner Grabbing
nc 192.168.100.8 8000

GET / HTTP/1.1
Host: test


# Discover Devices
sudo arp-scan --localnet

# Scan Phone
nmap <phone-ip>

# Notes
- localhost = your own device
- local IP = device inside local network
- 200 OK = request successful
- 400 Bad Request = invalid request
- banner grabbing = collecting service information
- headers = HTTP information
- wget downloads files
- curl sends requests
- nc creates raw connection to services
